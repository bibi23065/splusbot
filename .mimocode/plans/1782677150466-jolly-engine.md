# SplusBot: Cloudflare Worker Architecture Overhaul

## Goal
Replace the two-tier architecture (Cloudflare Worker → GitHub Actions Puppeteer) with a self-contained Cloudflare Worker that directly fetches Soroush Messenger web API via raw HTTP. Eliminate all Puppeteer/GitHub Actions dependencies.

---

## Phase A: Reverse Engineering Guide (User Action Required)

The Soroush Web API endpoints are not yet known. Before writing the Soroush client code, you must capture the actual API calls.

### Step-by-Step Instructions

1. **Open Chrome DevTools** → Network tab → Filter by `Fetch/XHR`
2. **Navigate to** `https://web.splus.ir/` and log in
3. **Capture these requests** while interacting with the app:
   - **Chat list load**: Look for requests to endpoints containing `chat`, `dialog`, `conversation`, or `folder`
   - **Message fetch**: Click a chat → look for requests containing `message`, `history`, or `get`
   - **User info**: Look for requests containing `user`, `me`, or `profile`
4. **For each request, record**:
   - Full URL (including query params)
   - HTTP method
   - Request headers (especially `Authorization`, `X-*`, `Content-Type`)
   - Request body (if POST)
   - Response body structure
5. **Check for request signing**: Look at the JavaScript source for any `sign`, `hash`, `hmac`, or `encrypt` functions applied to request bodies/headers
6. **Paste findings here** for code implementation

### Common Soroush API Patterns (from auth.py exploration)
- Base URLs: `api.splus.ir`, `wslb2.soroush-hamrah.ir`
- Auth headers likely include: `Authorization: Bearer <token>`, custom `X-Splus-*` headers
- Possible MTProto-over-HTTP or REST hybrid

### 5. `src/dedup.ts` — Deduplication Layer

```typescript
// src/dedup.ts
// Generates unique keys from message attributes and checks against KV

import { sha256 } from './cryptoHelper'; // or use SubtleCrypto

function generateMessageKey(chatId: number, text: string, timestamp: number): string {
  const payload = `${chatId}:${text}:${timestamp}`;
  // Simple hash using SubtleCrypto (available in Workers)
  return payload; // Will be hashed before KV lookup
}

async function hashString(str: string): Promise<string> {
  const encoder = new TextEncoder();
  const data = encoder.encode(str);
  const hashBuffer = await crypto.subtle.digest('SHA-256', data);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
}

export async function isDuplicate(kv: KVNamespace, chatId: number, text: string, timestamp: number): Promise<boolean> {
  const key = generateMessageKey(chatId, text, timestamp);
  const hash = await hashString(key);
  const kvKey = `splusbot:seen:${hash}`;
  const existing = await kv.get(kvKey);
  return existing !== null;
}

export async function markAsSeen(kv: KVNamespace, chatId: number, text: string, timestamp: number): Promise<void> {
  const key = generateMessageKey(chatId, text, timestamp);
  const hash = await hashString(key);
  const kvKey = `splusbot:seen:${hash}`;
  // Expire after 7 days to prevent unbounded growth
  await kv.put(kvKey, '1', { expirationTtl: 604800 });
}
```

### 6. `src/index.ts` — Main Worker (Complete Rewrite)

```typescript
import type { Env } from './types';
import { sendMsg, answerCallback, escapeHtml } from './telegram';
import { fetchUnreadMessages } from './soroush';
import { isDuplicate, markAsSeen } from './dedup';

async function getState(kv: KVNamespace, chatId: number) {
  const data = await kv.get(`splusbot:state:${chatId}`, 'json');
  return data ?? { state: 'UNAUTHENTICATED' };
}

async function setState(kv: KVNamespace, chatId: number, state: any) {
  await kv.put(`splusbot:state:${chatId}`, JSON.stringify(state));
}

export default {
  async fetch(request: Request, env: Env, ctx: ExecutionContext): Promise<Response> {
    if (request.method === 'GET') return new Response('SplusBot v4', { status: 200 });

    try {
      const update = await request.json() as any;
      const chatId = update.message?.chat?.id || update.callback_query?.message?.chat?.id;
      const text = update.message?.text || update.callback_query?.data || '';
      const callbackQueryId = update.callback_query?.id;

      if (!chatId || !env.KV) return new Response('ok', { status: 200 });
      if (callbackQueryId) await answerCallback(callbackQueryId, env.TELEGRAM_BOT_TOKEN);

      const state = await getState(env.KV, chatId);

      switch (state.state) {
        case 'UNAUTHENTICATED': {
          if (text === '/start') {
            const kb = { inline_keyboard: [[{ text: '🔑 Login to Soroush+', callback_data: 'login_splus' }]] };
            await sendMsg(chatId, 'Welcome! Click below to login.', kb, env.TELEGRAM_BOT_TOKEN);
          } else if (text === 'login_splus') {
            await setState(env.KV, chatId, { state: 'AWAITING_TOKEN' });
            const instructions = [
              'To get your session token:',
              '',
              '1. Open https://web.splus.ir in Chrome',
              '2. Log in with your phone number',
              '3. Press F12 → Application → Local Storage',
              '4. Find "user_auth" key and copy its value',
              '',
              'Paste the token below:',
            ].join('\n');
            await sendMsg(chatId, instructions, undefined, env.TELEGRAM_BOT_TOKEN);
          }
          break;
        }

        case 'AWAITING_TOKEN': {
          if (text && text.length > 10 && !text.startsWith('/')) {
            // Store token (could be raw string or JSON)
            let tokenData = text.trim();
            try { JSON.parse(text); } catch { /* keep as string */ }
            await env.KV.put(`splusbot:token:${chatId}`, tokenData);
            await setState(env.KV, chatId, { state: 'AUTHENTICATED' });
            const kb = { inline_keyboard: [[{ text: '📬 Check Messages', callback_data: 'check_now' }]] };
            await sendMsg(chatId, '✅ Authenticated! Click below to check unread messages.', kb, env.TELEGRAM_BOT_TOKEN);
          } else {
            await sendMsg(chatId, '⚠️ Invalid token. Paste the full session data.', undefined, env.TELEGRAM_BOT_TOKEN);
          }
          break;
        }

        case 'AUTHENTICATED': {
          if (text === 'check_now' || text === '/fetch') {
            const token = await env.KV.get(`splusbot:token:${chatId}`);
            if (!token) {
              await setState(env.KV, chatId, { state: 'UNAUTHENTICATED' });
              await sendMsg(chatId, '❌ Token expired. Send /start to re-login.', undefined, env.TELEGRAM_BOT_TOKEN);
              break;
            }

            await sendMsg(chatId, '🔄 Checking for unread messages...', undefined, env.TELEGRAM_BOT_TOKEN);

            try {
              const messages = await fetchUnreadMessages(token);

              if (messages.length === 0) {
                await sendMsg(chatId, '📭 No unread messages.', undefined, env.TELEGRAM_BOT_TOKEN);
                break;
              }

              // Dedup and format
              const newMessages: typeof messages = [];
              for (const msg of messages) {
                if (!await isDuplicate(env.KV, msg.chatId, msg.text, msg.timestamp)) {
                  await markAsSeen(env.KV, msg.chatId, msg.text, msg.timestamp);
                  newMessages.push(msg);
                }
              }

              if (newMessages.length === 0) {
                await sendMsg(chatId, '📭 No new unread messages (all previously seen).', undefined, env.TELEGRAM_BOT_TOKEN);
                break;
              }

              // Group by chat and send
              const grouped = newMessages.reduce((acc, m) => {
                (acc[m.chatTitle] ??= []).push(m);
                return acc;
              }, {} as Record<string, typeof messages>);

              for (const [chatTitle, msgs] of Object.entries(grouped)) {
                let text = `💬 *${escapeHtml(chatTitle)}* (${msgs.length} new)\n\n`;
                for (const m of msgs.slice(0, 10)) {
                  text += `👤 ${escapeHtml(m.senderName)}: ${escapeHtml(m.text.slice(0, 200))}\n`;
                }
                if (msgs.length > 10) text += `\n... and ${msgs.length - 10} more`;
                await sendMsg(chatId, text, undefined, env.TELEGRAM_BOT_TOKEN, 'Markdown');
              }

              const kb = { inline_keyboard: [[{ text: '📬 Check Again', callback_data: 'check_now' }]] };
              await sendMsg(chatId, `✅ Done! ${newMessages.length} new messages forwarded.`, kb, env.TELEGRAM_BOT_TOKEN);

            } catch (e: any) {
              const msg = e.message || 'Unknown error';
              if (msg.includes('401') || msg.includes('403') || msg.includes('Unauthorized')) {
                await setState(env.KV, chatId, { state: 'UNAUTHENTICATED' });
                await env.KV.delete(`splusbot:token:${chatId}`);
                await sendMsg(chatId, '❌ Session expired! Send /start to re-login.', undefined, env.TELEGRAM_BOT_TOKEN);
              } else {
                await sendMsg(chatId, `❌ Error: ${msg}`, undefined, env.TELEGRAM_BOT_TOKEN);
              }
            }
          } else if (text === '/logout') {
            await env.KV.delete(`splusbot:token:${chatId}`);
            await setState(env.KV, chatId, { state: 'UNAUTHENTICATED' });
            await sendMsg(chatId, '👋 Logged out. Send /start to login again.', undefined, env.TELEGRAM_BOT_TOKEN);
          }
          break;
        }
      }

      return new Response('ok', { status: 200 });
    } catch (e: any) {
      return new Response('ok', { status: 200 });
    }
  },
};
```
